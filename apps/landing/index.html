‚úÖ What AI generates with the SDK
import { Bpmn } from "@bpmn-sdk/core";

const defs = Bpmn.createProcess("my-flow")
  .startEvent("start")
  .serviceTask("task", {
    name: "Do Something",
    taskType: "my-worker",
  })
  .endEvent("end")
  .withAutoLayout()
  .build();

const xml = Bpmn.export(defs);
// ‚Üí valid BPMN 2.0 XML with full diagram

See it in action

Each code snippet below generates a complete, renderable BPMN diagram. Click the tabs to see different examples.

import { Bpmn } from "@bpmn-sdk/core";

const defs = Bpmn.createProcess("order-validation")
  .name("Order Validation")
  .startEvent("start", { name: "Order Received" })
  .serviceTask("validate", {
    name: "Validate Order",
    taskType: "validate-order",
  })
  .serviceTask("notify", {
    name: "Send Confirmation",
    taskType: "send-email",
  })
  .endEvent("end", { name: "Done" })
  .withAutoLayout()
  .build();

const xml = Bpmn.export(defs);
const defs = Bpmn.createProcess("approval-flow")
  .name("Approval Workflow")
  .startEvent("start", { name: "Request Submitted" })
  .serviceTask("review", {
    name: "Auto Review",
    taskType: "auto-review",
  })
  .exclusiveGateway("decision", { name: "Approved?" })
    .branch("approved", (b) =>
      b.condition("= approved = true")
        .serviceTask("process", {
          name: "Process Request",
          taskType: "process",
        })
        .endEvent("done", { name: "Completed" })
    )
    .branch("rejected", (b) =>
      b.defaultFlow()
        .serviceTask("reject", {
          name: "Send Rejection",
          taskType: "reject",
        })
        .endEvent("rejected", { name: "Rejected" })
    )
  .withAutoLayout()
  .build();
const defs = Bpmn.createProcess("order-fulfillment")
  .name("Order Fulfillment")
  .startEvent("start", { name: "Order Placed" })
  .parallelGateway("fork")
    .branch("payment", (b) =>
      b.serviceTask("pay", {
        name: "Process Payment",
        taskType: "payment-service",
      })
    )
    .branch("inventory", (b) =>
      b.serviceTask("stock", {
        name: "Reserve Inventory",
        taskType: "inventory-service",
      })
    )
    .branch("notifications", (b) =>
      b.serviceTask("email", {
        name: "Send Confirmation",
        taskType: "email-service",
      })
    )
  .serviceTask("ship", {
    name: "Ship Order",
    taskType: "shipping-service",
  })
  .endEvent("end", { name: "Order Complete" })
  .withAutoLayout()
  .build();
const defs = Bpmn.createProcess("ai-support-agent")
  .name("AI Support Agent")
  .startEvent("start", { name: "Ticket Created" })
  .serviceTask("enrich", {
    name: "Fetch Context",
    taskType: "fetch-customer-data",
  })
  .adHocSubProcess("agent", (sub) => {
    sub
      .serviceTask("search", {
        name: "Search KB",
        taskType: "knowledge-search",
      })
      .serviceTask("draft", {
        name: "Draft Response",
        taskType: "draft-reply",
      });
  }, {
    name: "AI Agent",
    taskDefinition: {
      type: "io.camunda:ai-agent:1",
    },
  })
  .exclusiveGateway("check", { name: "Confidence?" })
    .branch("high", (b) =>
      b.condition("= confidence > 0.9")
        .serviceTask("send", {
          name: "Auto-Reply",
          taskType: "send-reply",
        })
        .endEvent("resolved", { name: "Resolved" })
    )
    .branch("low", (b) =>
      b.defaultFlow()
        .userTask("escalate", {
          name: "Human Review",
          formId: "review-form",
        })
        .endEvent("escalated", { name: "Escalated" })
    )
  .withAutoLayout()
  .build();
